WebGL ist OpenGL
für das Internet.

Systemnahe, komplexe API:

In JavaScript produziert man rohe Daten für eine Szene; Dreiecke, Matrizen, usw.

Sie sind dann an Shadern geliefert, die ebenso in JavaScript produziert sind.

var vertexData = [
  // Position:
  0.0,  0.8,  0.0,  // (X, Y, Z)
  // Farbe (Rot):
  1.0, 0.0, 0.0, 1.0,  // (R, G, B, A)

  // Position:
  -0.8, -0.8,  0.0,
  // Farbe (Grün):
  0.0, 1.0, 0.0, 1.0,

  // Position:
  0.8, -0.8,  0.0,
  // Farbe (Blau):
  0.0, 0.0, 1.0, 1.0
];
// Vertex Shader
attribute vec3 positionAttr;
attribute vec4 colorAttr;

varying vec4 vColor;
void main(void) {
    gl_Position = vec4(positionAttr, 1.0);
    vColor = colorAttr;
}
// Fragment Shader
precision mediump float;

varying vec4 vColor;
void main(void) {
    gl_FragColor = vColor;
}
